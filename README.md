## 2\. –ê–Ω–∞–ª—ñ–∑ –º–µ—Ç–æ–¥—É –ó–∞—É—Ç–µ–Ω–¥–∞–π–∫–∞ (–ú–µ—Ç–æ–¥ –ü—Ä–∏–¥–∞—Ç–Ω–∏—Ö –ù–∞–ø—Ä—è–º–∫—ñ–≤)

### 2.1. –ü—Ä–∏–Ω—Ü–∏–ø —Ä–æ–±–æ—Ç–∏

–ú–µ—Ç–æ–¥ –ø–æ—Å–ª—ñ–¥–æ–≤–Ω–æ —Ä—É—Ö–∞—î—Ç—å—Å—è –≤ –Ω–∞–ø—Ä—è–º–∫—É, —â–æ –º–∞–∫—Å–∏–º—ñ–∑—É—î –∑–Ω–∏–∂–µ–Ω–Ω—è —Ñ—É–Ω–∫—Ü—ñ—ó, –ø—Ä–∏ —Ü—å–æ–º—É **–∑–∞–ª–∏—à–∞—é—á–∏ —Ä—ñ—à–µ–Ω–Ω—è –≤ –º–µ–∂–∞—Ö –¥–æ–ø—É—Å—Ç–∏–º–æ—ó –æ–±–ª–∞—Å—Ç—ñ**, –≤–∏–∑–Ω–∞—á–µ–Ω–æ—ó –æ–±–º–µ–∂–µ–Ω–Ω—è–º–∏. –ù–∞–ø—Ä—è–º–æ–∫ —Ä—É—Ö—É –≤–∏–∑–Ω–∞—á–∞—î—Ç—å—Å—è —Ä–æ–∑–≤'—è–∑–∞–Ω–Ω—è–º –¥–æ–ø–æ–º—ñ–∂–Ω–æ—ó –∑–∞–¥–∞—á—ñ –ª—ñ–Ω—ñ–π–Ω–æ–≥–æ –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—è.

### 2.2. –ü—Ä–∏–∫–ª–∞–¥: SLSQP —è–∫ –ê–Ω–∞–ª–æ–≥ –ó–∞—É—Ç–µ–Ω–¥–∞–π–∫–∞

–ù–∞ –ø—Ä–∞–∫—Ç–∏—Ü—ñ, –¥–ª—è –Ω–µ–ª—ñ–Ω—ñ–π–Ω–æ—ó –æ–ø—Ç–∏–º—ñ–∑–∞—Ü—ñ—ó –∑ –æ–±–º–µ–∂–µ–Ω–Ω—è–º–∏ –≤ Python —á–∞—Å—Ç–æ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î—Ç—å—Å—è **SLSQP** (Sequential Least Squares Programming), —è–∫–∏–π –∑–∞—Å–Ω–æ–≤–∞–Ω–∏–π –Ω–∞ —Å—Ö–æ–∂–∏—Ö –ø—Ä–∏–Ω—Ü–∏–ø–∞—Ö.

–ü—Ä–∏–∫–ª–∞–¥ –º—ñ–Ω—ñ–º—ñ–∑—É—î $f(x, y) = x^2 + y^2$ –∑–∞ —É–º–æ–≤–∏ –Ω–µ—Ä—ñ–≤–Ω–æ—Å—Ç—ñ $x + y^2 \ge 1$.

```python
import numpy as np
from scipy.optimize import minimize

def objective(x):
    return x[0]**2 + x[1]**2

def constraint1(x):
    return x[0] + x[1]**2 - 1 # –ü–æ–≤–∏–Ω–Ω–æ –±—É—Ç–∏ G(x) >= 0

con = ({'type': 'ineq', 'fun': constraint1})
x0 = np.array([0.5, 0.5])
result_slsqp = minimize(objective, x0, method='SLSQP', constraints=con)

print("--- –ê–Ω–∞–ª—ñ–∑ –ú–µ—Ç–æ–¥—É –ó–∞—É—Ç–µ–Ω–¥–∞–π–∫–∞ (–ß–µ—Ä–µ–∑ SLSQP –∑ –æ–±–º–µ–∂–µ–Ω–Ω—è–º–∏) ---")
print(f"–ó–Ω–∞–π–¥–µ–Ω–∏–π –º—ñ–Ω—ñ–º—É–º x: {result_slsqp.x.round(4)}")
print(f"–ú—ñ–Ω—ñ–º–∞–ª—å–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è —Ñ—É–Ω–∫—Ü—ñ—ó: {result_slsqp.fun.round(4)}")
print(f"–¢–µ—Å—Ç –æ–±–º–µ–∂–µ–Ω–Ω—è (–º–∞—î –±—É—Ç–∏ >= 0): {constraint1(result_slsqp.x).round(4)}")
```
